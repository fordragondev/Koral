```typescript
// app/layout.tsx — Root layout: per-tenant theme injection + FOUC prevention (Pitfall 7)
import type { Metadata } from 'next';
import { getTenant } from '@/server/tenant/resolve';
import { auth } from '@/server/auth/config';
import '@/styles/globals.css';

export async function generateMetadata(): Promise<Metadata> {
  const tenant = await getTenant();
  return {
    title: tenant.name,
    description: tenant.tagline ?? tenant.name,
    icons: tenant.faviconUrl ? [{ url: tenant.faviconUrl }] : undefined,
  };
}

export default async function RootLayout({ children }: { children: React.ReactNode }) {
  const [tenant, session] = await Promise.all([getTenant(), auth()]);
  // themeLight / themeDark are stored as jsonb — Drizzle returns them as plain objects
  const light = (typeof tenant.themeLight === 'string'
    ? JSON.parse(tenant.themeLight)
    : tenant.themeLight) as Record<string, string>;
  const dark = (typeof tenant.themeDark === 'string'
    ? JSON.parse(tenant.themeDark)
    : tenant.themeDark) as Record<string, string>;

  const css = `
    :root {
      --color-bg: ${light.bg};
      --color-surface: ${light.surface};
      --color-accent: ${light.accent};
      --color-accent-hover: ${light.accentHover};
      --color-fg: ${light.fg};
      --color-primary: ${light.primary};
    }
    [data-theme="dark"] {
      --color-bg: ${dark.bg};
      --color-surface: ${dark.surface};
      --color-accent: ${dark.accent};
      --color-accent-hover: ${dark.accentHover};
      --color-fg: ${dark.fg};
      --color-primary: ${dark.primary};
    }
  `;

  // D-30: if the signed-in user has a known darkMode preference, use it as the server-side
  // initial value. This enables cross-device restore — on a new device the session carries
  // the preference so localStorage (which starts empty) is not needed.
  // null/undefined → client-side FOUC script takes over (OS preference or localStorage).
  const userDarkMode: boolean | null = (session as any)?.user?.darkMode ?? null;
  const serverTheme = userDarkMode === true ? 'dark' : userDarkMode === false ? 'light' : null;

  // Inline synchronous script — MUST run before paint to prevent dark mode flash (Pitfall 7)
  // D-28: default to OS preference if localStorage empty. D-31: data-theme attribute on <html>.
  // If serverTheme is non-null (signed-in user with known preference), it takes priority over
  // localStorage so the correct theme renders on page load on a new device.
  const themeInitScript = serverTheme
    ? `(function(){try{var t=${JSON.stringify(serverTheme)};localStorage.setItem('theme',t);document.documentElement.setAttribute('data-theme',t);}catch(e){}})();`
    : `(function(){try{var t=localStorage.getItem('theme')||(window.matchMedia('(prefers-color-scheme: dark)').matches?'dark':'light');document.documentElement.setAttribute('data-theme',t);}catch(e){}})();`;

  return (
    <html suppressHydrationWarning>
      <head>
        <style dangerouslySetInnerHTML={{ __html: css }} />
        <script dangerouslySetInnerHTML={{ __html: themeInitScript }} />
      </head>
      <body>{children}</body>
    </html>
  );
}
```

// i18n/request.ts — next-intl server-side request config
import { getRequestConfig } from 'next-intl/server';
import { routing } from './routing';

export default getRequestConfig(async ({ requestLocale }) => {
  let locale = await requestLocale;
  if (!locale || !routing.locales.includes(locale as any)) {
    locale = routing.defaultLocale;
  }
  const [common, nav, auth] = await Promise.all([
    import(`../messages/${locale}/common.json`).then((m) => m.default),
    import(`../messages/${locale}/nav.json`).then((m) => m.default),
    import(`../messages/${locale}/auth.json`).then((m) => m.default),
  ]);
  return {
    locale,
    messages: { common, nav, auth },
  };
});
```

// app/[locale]/layout.tsx — locale-aware app shell
import { NextIntlClientProvider } from 'next-intl';
import { getMessages, setRequestLocale } from 'next-intl/server';
import { notFound } from 'next/navigation';
import { routing } from '@/i18n/routing';

export function generateStaticParams() {
  return routing.locales.map((locale) => ({ locale }));
}

export default async function LocaleLayout({
  children,
  params,
}: {
  children: React.ReactNode;
  params: Promise<{ locale: string }>;
}) {
  const { locale } = await params;
  if (!routing.locales.includes(locale as any)) notFound();
  setRequestLocale(locale);
  const messages = await getMessages();
  return (
    <NextIntlClientProvider messages={messages} locale={locale}>
      <div lang={locale}>{children}</div>
    </NextIntlClientProvider>
  );
}
```

Note: The `<html lang>` attribute is set dynamically by the locale layout wrapping its children in a `<div lang={locale}>`. next-intl v4 does not mutate `<html>` from a nested layout — the test asserts the `<div lang>` attribute instead. <threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| DB → root layout | `getTenant()` reads jsonb theme columns. A malicious theme value (e.g. containing `</style>`) could break out of the `<style>` block and inject script. Theme values are written only by trusted seed/admin paths, but the layout must still treat them as untrusted data |
| Root layout → inline script | The theme-init script is rendered with `dangerouslySetInnerHTML`. Its content is a constant literal — safe — but any future edit that interpolates tenant data into this script would be a XSS sink |
| Tenant config → CSS custom properties | CSS custom property values are inserted into a `<style>` block. CSS injection is possible if a theme value contains `}`, `;`, or `<` |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Tampering | CSS injection via malicious tenant theme value (e.g. `bg: "#000; } body { display:none"`) | mitigate | Theme values are set only by the seed script (Plan 03) and a future admin flow (post-Phase 1). For Phase 1: the seed script hard-codes safe hex values; no user-facing theme editor exists. Accepted as deferred hardening: a future admin editor must validate theme values with a regex `/^#[0-9a-f]{6}$/i` before writing. |
| T-05-02 | Cross-site Scripting | Malicious tenant name or tagline reflected into `<title>` or `<h1>` | mitigate | React automatically escapes text content in JSX, so `{tenant.name}` and `{tenant.tagline}` are safe. The `<style>` block uses `dangerouslySetInnerHTML` but only with hex color strings, not tenant text. |
